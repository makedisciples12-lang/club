📊 Google Sheets 연동 (Apps Script)

아래 Apps Script 코드를 Google Sheets에 배포하면 학생 제출 데이터가 자동 저장됩니다.

function doPost(e) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); var data = JSON.parse(e.postData.contents); var row = [ new Date(), data.학생이름, data.팀번호, data.텍스트구간, data.사전Q1||'', data.사전Q2||'', data.사전Q3||'', data.사전Q4||'', data.사전Q5||'', data.사전Q6||'', data.사전Q7||'', data.사전Q8||'', data.총발화수||0, data.C발화수||0, data.Q발화수||0, data.S발화수||0, data.코딩결과JSON||'', data.사후Q1||'', data.사후Q2||'', data.사후Q3||'', data.사후Q4||'', data.사후Q5||'', data.사후Q6||'', data.사후Q7||'', data.사후Q8||'' ]; sheet.appendRow(row); return ContentService.createTextOutput( JSON.stringify({result:'success'}) ).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return ContentService.createTextOutput('OK'); }

배포 방법: Google Sheets → 확장프로그램 → Apps Script → 위 코드 붙여넣기 → 배포 → 웹앱으로 배포 (모든 사용자 접근)
